#include "ift.h"
int main(int argc, const char *argv[]) {
if (argc != 2)
iftExit("iftTestCSV <csv_file.csv", "main");
char csv_pathname[512];
strcpy(csv_pathname, argv[1]);
iftCSV *csv = iftReadCSV(csv_pathname, ',');
iftPrintCSV(csv);
iftDestroyCSV(&csv);
return 0;
}
